### CRC 在线计算（Python）
```python
def calc_crc8(data):
    crc = 0x00
    for byte in data:
        crc ^= byte
        for _ in range(8):
            if crc & 0x80:
                crc = (crc << 1) ^ 0x07
            else:
                crc <<= 1
        crc &= 0xFF
    return crc

# 示例
payload = [0x01, 0x01]  # CMD=0x01, DATA=0x01
print(f"CRC-8: 0x{calc_crc8(payload):02X}")
```
